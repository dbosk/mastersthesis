\section{Inledning}
\begin{frame}{Bakgrund}
	\begin{itemize}
		\uncover<2>{\item Bara räkning, ingen matematik}
		\uncover<3-5>{\item \citet{Skolinspektionen2010uim} granskade
			matematikundervisningen i kursen Matematik A}
			\begin{itemize}
				\uncover<4>{\item Lektionen består av två delar:
					gemensam genomgång och eget arbete}
				\uncover<5>{\item Dominerande arbetsformen är enskilt arbete i
					läroboken}
			\end{itemize}
	\end{itemize}
\end{frame}
\begin{frame}{Syfte}
	\begin{itemize}
		\item<2>
			Lägga grunden för en formalisering av gymnasiematematiken
		\item<3>
			Ta fram en början till ett undervisningsmaterial
		\item<4>
			Undersöka hur detta kan användas
	\end{itemize}
\end{frame}
\begin{frame}{Frågeställning}
	\begin{enumerate}
		\uncover<2-4>{\item\label{rq:Tillåtande}
			På vilka sätt tillåter, eller hindrar, gymnasieskolans nya
			ämnesplan i matematik att undervisa formell matematik?}
			\begin{enumerate}
				\uncover<3>{\item\label{rq:MängdInnehåll}
					Hur mycket extra innehåll behövs uppskattningsvis för att
					undervisa Matematik 1c med formell matematik?}
				\uncover<4>{\item\label{rq:Övningsuppgifter}
					Vilken typ av elevarbete, om något särskilt, kräver
					ämnesplanen?}
			\end{enumerate}
		\uncover<5>{\item\label{rq:Material}
			På vilka sätt kan ett undervisningsmaterial för formell matematik
			motsvarande Matematik 1c se ut?}
		\uncover<6>{\item\label{rq:Möjligt}
			Hur är möjligheterna att undervisa formell matematik på
			gymnasienivå?}
\end{enumerate}
\end{frame}
